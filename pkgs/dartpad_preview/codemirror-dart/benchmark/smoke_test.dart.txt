mixin Swimmer {
  bool canSwim = true;
  void swim() => print('Swimming!');
}

class Dog extends Animal with Swimmer implements Comparable<Dog> {
  final String breed;
  int _age;

  // Constructors
  Dog(this.breed, this._age);
  Dog.puppy(this.breed) : _age = 0;

  // Factory constructor
  factory Dog.fromMap(Map<String, dynamic> map) {
    return Dog(map['breed'], map['age']);
  }

  // Getters & Setters
  int get age => _age;
  set age(int value) {
    if (value >= 0) _age = value;
  }

  @override
  void speak() {
    print('Woof!');
    super.canSwim;
  }

  @override
  int compareTo(Dog other) => _age.compareTo(other._age);
}

// --- Control Flow & Pattern Matching (Dart 3) ---
void controlFlowSyntax(Object obj) {
  // If / Else
  if (obj is int) {
    print('Integer');
  } else if (obj is String) {
    print('String');
  } else {
    print('Unknown');
  }

  // For loops
  for (int i = 0; i < 10; i++) {
    if (i == 5) continue;
    if (i == 8) break;
  }

  for (final item in numbers) {
    print(item);
  }

  // While / Do-While
  int counter = 0;
  while (counter < 5) {
    counter++;
  }

  do {
    counter--;
  } while (counter > 0);

  // Switch statement with Pattern Matching
  switch (obj) {
    case int n when n > 0:
      print('Positive Int');
    case String s:
      print('String of length ${s.length}');
    case [int a, int b]:
      print('List of two ints: $a, $b');
    case {'name': String name}:
      print('Map with name: $name');
    case Dog(breed: 'Poodle', age: var a):
      print('Poodle of age $a');
    case _:
      print('Default case');
  }

  // Try / Catch / Finally
  try {
    throw Exception('Error!');
  } on Exception catch (e, stackTrace) {
    print('Caught: $e');
  } catch (e) {
    rethrow;
  } finally {
    print('Cleanup');
  }
}
